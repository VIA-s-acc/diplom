126 def update_cell(self, r, c, x, w, v):
127 """!
128
129 Update the moisture level of a cell in the field
130
131 @param r Row index of the cell.
132 @param c Column index of the cell.
133 @param x Current position of watering machine.
134 @param w Watering intensity.
135 @param v Water level.
136
137 @return float
138 @details Updates the moisture level of a cell in the field based on the current position of the watering machine, the watering intensity, and the water level.
139
140 """
141 if self.field[r][c] == -1:
142 return 0
143 d_rc = ((r - self.line) ** 2 + (c - x) ** 2) ** 0.5
144 term = w * (self.Wm * self.Deltat / (d_rc**2 + 1)**self.beta) * np.exp(-self.alpha * v)
145 self.field[r][c] += term
146 return term

149 """
150 @param x Current position of watering machine.

152 @param v Water level.

155
156 """
157 start_col = max(0, x - self.rx_cells)
158 end_col = min(self.cols, x + 1)
